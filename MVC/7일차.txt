 # POST 매핑과 Redirection.

request로 method 확인하기.
if ( request.getMethod( ).equals( "POST 또는 GET" ) ) { 실행할 내용; }

어노테이션으로 POST 매핑.
@RequestMapping( value="urlMapping" , method=RequestMethod.GET 또는 POST ) - 옛날 방식.
@GetMapping( "urlMapping" ) - 요즘 방식.
@PostMapping( "urlMapping" ) - 요즘 방식.

redirect 설정하기.
return "redirect:url" ;

forward 설정하기.
return "포워드 될 url" ;
